    def __init__(self, numbers):
        Thread.__init__(self)
        self.numbers = numbers

    def run(self):
        n = len(self.numbers)
        for i in range(0, n):
            for j in range(0, n - i - 1):

                    self.numbers[j] = self.numbers[j + 1]
                    self.numbers[j + 1] = temp
